nls_add_runtime_target(
  NAME NLS_Base
  SOURCE_DIR ${CMAKE_CURRENT_SOURCE_DIR}
  LIBS
    json11
  APPEND_INCLUDE_DIR
    ${NLS_ThirdParty_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}/Generated
)

set(NLS_META_GEN_DIR "${CMAKE_CURRENT_BINARY_DIR}/Generated")
set(NLS_META_GEN_CPP "${NLS_META_GEN_DIR}/MetaGenerated.cpp")
set(NLS_META_GEN_H "${NLS_META_GEN_DIR}/MetaGenerated.h")

set(NLS_META_PARSER_LD_LIBRARY_PATH "$ENV{LD_LIBRARY_PATH}")
if(EXISTS "/usr/lib/llvm-10/lib")
  if(NLS_META_PARSER_LD_LIBRARY_PATH)
    set(NLS_META_PARSER_LD_LIBRARY_PATH "/usr/lib/llvm-10/lib:${NLS_META_PARSER_LD_LIBRARY_PATH}")
  else()
    set(NLS_META_PARSER_LD_LIBRARY_PATH "/usr/lib/llvm-10/lib")
  endif()
endif()

file(GLOB_RECURSE NLS_RUNTIME_HEADERS
  "${NLS_ROOT_DIR}/Runtime/*.h"
  "${NLS_ROOT_DIR}/Runtime/*.hpp"
  "${NLS_ROOT_DIR}/Runtime/*.inl"
)

set(NLS_META_MARKED_HEADERS)
foreach(_hdr IN LISTS NLS_RUNTIME_HEADERS)
  if(_hdr MATCHES "/Generated/" OR _hdr MATCHES "/Reflection/Macros\.h$")
    continue()
  endif()

  file(READ "${_hdr}" _hdr_content)

  # 仅对带反射标记声明的头文件生成 .gen.cpp；无标记不生成
  if((_hdr_content MATCHES "Meta[ \t\r\n]*\\([^\\)]*\\)[ \t\r\n]*(class|struct)" OR
      _hdr_content MATCHES "CLASS[ \t\r\n]*\\(" OR
      _hdr_content MATCHES "STRUCT[ \t\r\n]*\\(" OR
      _hdr_content MATCHES "(class|struct)[^\n]*annotate[ \t\r\n]*\\(")
      AND _hdr_content MATCHES "(NLS::meta::Object|[^A-Za-z0-9_]Object[^A-Za-z0-9_])")
    list(APPEND NLS_META_MARKED_HEADERS "${_hdr}")
  endif()
endforeach()

set(NLS_META_GEN_PARTS)
set(NLS_META_REGISTER_FUNCS)

foreach(_hdr IN LISTS NLS_META_MARKED_HEADERS)
  file(RELATIVE_PATH _hdr_rel "${NLS_ROOT_DIR}/Runtime" "${_hdr}")
  string(REPLACE "\\" "/" _hdr_rel "${_hdr_rel}")
  string(MAKE_C_IDENTIFIER "${_hdr_rel}" _hdr_id)

  set(_gen_cpp "${NLS_META_GEN_DIR}/${_hdr_id}.gen.cpp")
  set(_gen_func "RegisterReflectionTypes_${_hdr_id}")

  add_custom_command(
    OUTPUT "${_gen_cpp}"
    COMMAND ${CMAKE_COMMAND} -E make_directory "${NLS_META_GEN_DIR}"
    COMMAND ${CMAKE_COMMAND} -E env "LD_LIBRARY_PATH=${NLS_META_PARSER_LD_LIBRARY_PATH}" "${NLS_META_PARSER_DOTNET}" run --project "${NLS_ROOT_DIR}/Tools/MetaParser/src/MetaParser.csproj" --configuration "$<CONFIG>" --no-build --
            --root "${NLS_ROOT_DIR}"
            --header "${_hdr}"
            --out "${_gen_cpp}"
            --func "${_gen_func}"
    DEPENDS
            MetaParser
            "${_hdr}"
            "${NLS_ROOT_DIR}/Tools/MetaParser/src/Main.cs"
            "${NLS_ROOT_DIR}/Tools/MetaParser/src/Templates/MetaGenerated.cpp.tt"
    VERBATIM
  )

  list(APPEND NLS_META_GEN_PARTS "${_gen_cpp}")
  list(APPEND NLS_META_REGISTER_FUNCS "${_gen_func}")
endforeach()

file(WRITE "${NLS_META_GEN_H}" "#pragma once\n\n#include \"Reflection/ReflectionDatabase.h\"\n\nnamespace NLS::meta_generated\n{\nvoid RegisterReflectionTypes(NLS::meta::ReflectionDatabase& db);\nvoid RegisterReflectionTypes();\n}\n")

file(WRITE "${NLS_META_GEN_CPP}" "// Auto-generated by CMake (MetaParser aggregator)\n#include \"MetaGenerated.h\"\n\nnamespace NLS::meta_generated\n{\n")
foreach(_func IN LISTS NLS_META_REGISTER_FUNCS)
  file(APPEND "${NLS_META_GEN_CPP}" "void ${_func}(NLS::meta::ReflectionDatabase& db);\n")
endforeach()

file(APPEND "${NLS_META_GEN_CPP}" "\nvoid RegisterReflectionTypes(NLS::meta::ReflectionDatabase& db)\n{\n")
foreach(_func IN LISTS NLS_META_REGISTER_FUNCS)
  file(APPEND "${NLS_META_GEN_CPP}" "    ${_func}(db);\n")
endforeach()

file(APPEND "${NLS_META_GEN_CPP}" "}\n\nvoid RegisterReflectionTypes()\n{\n    auto& db = NLS::meta::ReflectionDatabase::Instance();\n    RegisterReflectionTypes(db);\n}\n}\n")

add_custom_target(NLS_MetaParser_Generate DEPENDS ${NLS_META_GEN_PARTS})
add_dependencies(NLS_Base NLS_MetaParser_Generate)

target_sources(NLS_Base PRIVATE "${NLS_META_GEN_CPP}" ${NLS_META_GEN_PARTS})
